#include<iostream>
using namespace std;

int main(){
    int age;
    cout<<"Tell me your age"<<endl;
    cin>>age;

    // if , else if ,else ladder
    if ((age<18) && (age>0)){
        cout<<"you cannot come to party"<<endl;
    }
    else if (age==18){
        cout<<"You can come to the party"<<endl;
    }
    else if (age<=0)
    {
        cout<<"You are not yet born!"<<endl;
    }
    
    else{
        cout<<"You can to the party"<<endl;
    }
    
    //switch case
    switch (age){
    case 18:
        cout<<"You are 18"<<endl;
        break;

    case 22:
        cout<<"You are 22"<<endl;
        break;

    case 2:
        cout<<"You are 18"<<endl;
        break;
    
    default:
        cout<<"No special cases"<<endl;
        break;
    }
    
    return 0;
}

#include<iostream>

int main(){
    // waht is a pointers? ---> it is a data types which holds the address of other data types
    int a=3;
    int* b = &a;
    // & ---> (address of) operators
    cout<<"The address of a is "<<&a<<endl;
    cout<<"The address of a is "<<b<<endl;

    // * ---> (value at) derefence operator
    cout<<"The value of addres b "<<*b<<endl;

    // pointer to pointer 
    int** c = &b;
    cout<<"The address of b is "<<&b<<endl;
    cout<<"The address of b is "<<c<<endl;
    cout<<"The value at address c is "<<*c<<endl;
    cout<<"The value at address value_at(value_at(c)) is "<<**c<<endl;
    return 0;
}
